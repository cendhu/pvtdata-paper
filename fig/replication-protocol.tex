% ============ (a) Order-Execute-Commit ============
\begin{scope}[local bounding box=protocola]
  \def\colC{0}  \def\colA{1.6}  \def\colB{2.8}  \def\colN{4.0}  \def\colO{5.8}

  \node[hdr] at (\colC,0) {Client};
  \node[hdr] at (\colA,0) {Peer 1};
  \node[hdr] at (\colB,0) {Peer 2};
  \node[hdr] at (\colN,0) {Peer n};
  \node[hdr, text width=1.3cm, align=center, font=\normalsize] at (\colO,0) {Ordering\\[-1pt]Service};

  \foreach \x in {\colC,\colA,\colB,\colN,\colO}
    \draw[lifeline] (\x,-0.2) -- (\x,-3.0);

  % Step 1: Client -> Ordering Service
  \draw[msg] (\colC,-0.45) -- node[above, pos=0.12] {$T_i$} (\colO,-0.45);
  \node at (\colC-0.35,-0.45) {\cstep{1}};

  % Step 2: Block creation
  \node at ($({\colN},0)!0.75!({\colO},0)+(0,-0.92)$) {\cstep{2}};
  \node[align=center, font=\normalsize, anchor=west] at ($({\colN},0)!0.75!({\colO},0)+(0.15,-0.92)$) {block\\[-1pt]creation};

  % Step 3: Broadcast block
  \draw[bcast] (\colO,-1.35) -- node[above] {block} (\colA,-1.35);
  \draw[bcast] (\colO,-1.35) -- (\colB,-1.35);
  \draw[bcast] (\colO,-1.35) -- (\colN,-1.35);
  \node at (\colA-0.35,-1.35) {\cstep{3}};

  % Step 4: Execution
  \node at (\colA-0.35,-1.95) {\cstep{4}};
  \node at (\colN+0.35,-1.95) {\cstep{4}};
  \draw[decorate,decoration={brace,amplitude=2pt,mirror}] (\colA-0.1,-1.7) -- (\colN+0.1,-1.7)
    node[midway,below=2pt] {execution};

  % Step 5: Commit
  \node at (\colA-0.35,-2.6) {\cstep{5}};
  \node at (\colN+0.35,-2.6) {\cstep{5}};
  \draw[decorate,decoration={brace,amplitude=2pt,mirror}] (\colA-0.1,-2.35) -- (\colN+0.1,-2.35)
    node[midway,below=2pt] {commit};

  % Legend (single line)
  \node[draw,rounded corners,fill=white,inner sep=2pt,anchor=north west,font=\normalsize] (lega) at (\colC-0.2,-3.05) {%
    \cstep{1}\,\cstep{2}\,\cstep{3}\;Ordering\quad \cstep{4}\;Execution\quad \cstep{5}\;Commit%
  };
\end{scope}
\node[below=0.02cm,font=\normalsize, align=center] at (protocola.south) {(a) Order-Execute-Commit\\[-1pt]{\footnotesize (Ethereum~\cite{ethereum}, Quorum~\cite{Quorum}, Besu~\cite{besu})}};


% ============ (b) Execute-Order-In-Parallel & Commit ============
\begin{scope}[xshift=7.8cm, local bounding box=protocolb]
  \def\colC{0}  \def\colA{1.6}  \def\colB{2.8}  \def\colN{4.0}  \def\colO{5.8}

  \node[hdr] at (\colC,0) {Client};
  \node[hdr] at (\colA,0) {Peer 1};
  \node[hdr] at (\colB,0) {Peer 2};

  \node[hdr, text width=1.3cm, align=center, font=\normalsize] at (\colO,0) {Ordering\\[-1pt]Service};

  \foreach \x in {\colC,\colA,\colB,\colN,\colO}
    \draw[lifeline] (\x,-0.2) -- (\x,-3.0);

  % Step 1: Client -> All nodes AND Ordering Service
  \draw[msg] (\colC,-0.45) -- node[above] {$T_i$} (\colA,-0.45);
  \draw[msg] (\colC,-0.45) -- (\colB,-0.45);

  \draw[msg] (\colC,-0.45) -- (\colO,-0.45);

  % Step 2: Execution on peers (parallel with ordering)
  \node at (\colA-0.35,-1.0) {\cstep{2}};
  \node at (\colN+0.35,-1.0) {\cstep{2}};
  \draw[decorate,decoration={brace,amplitude=2pt,mirror}] (\colA-0.1,-0.75) -- (\colN+0.1,-0.75)
    node[midway,below=2pt] {execution};

  % Step 2: Ordering (parallel with execution)
  \node at ($({\colN},0)!0.75!({\colO},0)+(0,-0.92)$) {\cstep{2}};
  \node[align=center, font=\normalsize, anchor=west] at ($({\colN},0)!0.75!({\colO},0)+(0.15,-0.92)$) {block\\[-1pt]creation};

  % Step 3: Block broadcast
  \draw[bcast] (\colO,-1.45) -- node[above] {block} (\colA,-1.45);
  \draw[bcast] (\colO,-1.45) -- (\colB,-1.45);
  \draw[bcast] (\colO,-1.45) -- (\colN,-1.45);
  \node at (\colA-0.35,-1.45) {\cstep{3}};

  % Step 4: Commit
  \node at (\colA-0.35,-2.1) {\cstep{4}};
  \node at (\colN+0.35,-2.1) {\cstep{4}};
  \draw[decorate,decoration={brace,amplitude=2pt,mirror}] (\colA-0.1,-1.85) -- (\colN+0.1,-1.85)
    node[midway,below=2pt] {commit};

  % Legend (single line)
  \node[draw,rounded corners,fill=white,inner sep=2pt,anchor=north west,font=\normalsize] at (\colC-0.2,-3.05) {%
    \cstep{1}\,\cstep{2}\;Execution \& Ordering\quad \cstep{3}\;Block Delivery\quad \cstep{4}\;Commit%
  };
\end{scope}
\node[below=0.02cm,font=\normalsize, align=center] at (protocolb.south) {(b) Execute-Order-In-Parallel \& Commit\\[-1pt]{\footnotesize (BlockchainRDBMS~\cite{pgchain})}};
